## Differences in sequences 

Determined differences between original sequences and latest sequences from NCBI using [edlib] global alignment edit distances.

### Code

```python
# Download latest sequence from NCBI using Entrez API
recs = {}
for acc in tqdm(df.chrom.unique()):
    with Entrez.efetch(db='nucleotide', id=acc, rettype='gb', retmode='text', ) as h:
        recs[acc] = list(SeqIO.parse(h, 'genbank'))
# Parse original CSFV sequences used to create AmpliSeq panel
orig_recs = {rec.id: rec for rec in SeqIO.parse('CSFV_AmpliSeq.fasta', 'fasta')}
# Print differences between original sequences and latest sequences from NCBI
CHAR_PER_LINE = 80
for rid in recs.keys():
    orig_rec = orig_recs[rid]
    ncbi_rec = recs[rid][0]
    o_seq = str(orig_rec.seq)
    n_seq = str(ncbi_rec.seq)
    if o_seq != n_seq:
        alignment = edlib.align(o_seq, n_seq, task='path')
        print('='*CHAR_PER_LINE)
        print(f'{rid} | accession={ncbi_rec.id} | {alignment}')
        nice_aln = edlib.getNiceAlignment(alignment, o_seq, n_seq)
        q_aln = nice_aln['query_aligned']
        match_aln = nice_aln['matched_aligned']
        t_aln = nice_aln['target_aligned']
        for i in range(0, len(q_aln), CHAR_PER_LINE):
            q_snippet = q_aln[i:(i+CHAR_PER_LINE+1)]
            m_snippet = match_aln[i:(i+CHAR_PER_LINE+1)]
            t_snippet = t_aln[i:(i+CHAR_PER_LINE+1)]
            if q_snippet != t_snippet:
                for x in (q_snippet, m_snippet, t_snippet):
                    print(f'{i: 6} {x}')
                print()
        print()
```
